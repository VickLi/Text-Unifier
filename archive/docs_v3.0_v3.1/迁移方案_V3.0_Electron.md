```typescript
import { app, BrowserWindow, dialog, ipcMain } from 'electron';
import * as path from 'path';
import * as fs from 'fs';

// 加载 napi-rs 编译的原生模块
const nativeModule = require('../native/index.node');

let mainWindow: BrowserWindow | null = null;

function createWindow() {
    mainWindow = new BrowserWindow({
        width: 1200,
        height: 800,
        minWidth: 900,
        minHeight: 600,
        title: '文档终版确定器 - Text Unifier',
        webPreferences: {
            preload: path.join(__dirname, 'preload.js'),
            contextIsolation: true,
            nodeIntegration: false,
        },
    });

    // 开发模式加载 Vite dev server，生产模式加载打包文件
    if (process.env.NODE_ENV === 'development') {
        mainWindow.loadURL('http://localhost:1420');
    } else {
        mainWindow.loadFile(path.join(__dirname, '../dist/index.html'));
    }

    mainWindow.setMenuBarVisibility(false);
}

// ═══════════════════════════════════════════
// IPC 处理器：将所有 napi 调用封装为 IPC
// ═══════════════════════════════════════════

/** 扫描并分析文件 */
ipcMain.handle('scan-files', async (_event, paths: string[]) => {
    // 验证路径
    const validPaths = paths.filter(p => {
        try {
            return fs.existsSync(p) && p.toLowerCase().endsWith('.txt');
        } catch {
            return false;
        }
    });
    if (validPaths.length === 0) {
        throw new Error('没有有效的 .txt 文件');
    }
    return nativeModule.scanFiles(validPaths);
});

/** 文档排版（去硬回车） */
ipcMain.handle('format-document', async (_event, text: string) => {
    return nativeModule.formatDocument(text);
});

/** 导出文件（打开保存对话框 + 写入） */
ipcMain.handle('export-file', async (_event, paragraphs: string[]) => {
    if (!mainWindow) throw new Error('窗口未初始化');

    const { canceled, filePath } = await dialog.showSaveDialog(mainWindow, {
        defaultPath: 'Merged_Document.txt',
        filters: [{ name: 'Text Files', extensions: ['txt'] }],
    });

    if (canceled || !filePath) {
        throw new Error('用户取消了保存');
    }

    const content = paragraphs.join('\n\n');
    fs.writeFileSync(filePath, content, 'utf-8');

    return { saved_path: filePath };
});

/** 选择 .txt 文件对话框 */
ipcMain.handle('select-files', async () => {
    if (!mainWindow) throw new Error('窗口未初始化');

    const { canceled, filePaths } = await dialog.showOpenDialog(mainWindow, {
        filters: [{ name: 'Text Files', extensions: ['txt'] }],
        properties: ['openFile', 'multiSelections'],
    });

    if (canceled) return [];

    // 收集文件信息
    return filePaths.map(fp => {
        const stats = fs.statSync(fp);
        return {
            name: path.basename(fp),
            path: fp,
            size: stats.size,
        };
    });
});

// ═══════════════════════════════════════════
// 应用生命周期
// ═══════════════════════════════════════════

app.whenReady().then(createWindow);

app.on('window-all-closed', () => {
    if (process.platform !== 'darwin') app.quit();
});

app.on('activate', () => {
    if (BrowserWindow.getAllWindows().length === 0) createWindow();
});
```

```typescript
/**
 * IPC 通信层（V3.0：Electron contextBridge）
 *
 * 所有 IPC 调用通过 window.electronAPI 桥接，
 * 由 electron/preload.ts 注入。
 * 保留了指数退避重试逻辑与超时控制。
 */

// 最大重试次数
const MAX_RETRIES = 3;
const RETRY_BASE_MS = 500;

/** 带指数退避重试的包装器 */
async function withRetry<T>(fn: () => Promise<T>, retries = MAX_RETRIES): Promise<T> {
    for (let attempt = 0; attempt <= retries; attempt++) {
        try {
            return await fn();
        } catch (error) {
            if (attempt === retries) throw error;
            const delay = RETRY_BASE_MS * Math.pow(2, attempt);
            console.warn(`IPC 调用失败，${delay}ms 后重试 (${attempt + 1}/${retries}):`, error);
            await new Promise(resolve => setTimeout(resolve, delay));
        }
    }
    throw new Error('IPC 调用失败: 重试次数耗尽');
}

/** 获取 electronAPI 引用 */
function getAPI() {
    const api = (window as any).electronAPI;
    if (!api) {
        throw new Error('electronAPI 未初始化 — 请确保通过 Electron 启动应用');
    }
    return api;
}

/** 扫描并分析文件 */
export async function scanFiles(paths: string[]): Promise<AnalysisReport> {
    return withRetry(() => getAPI().scanFiles(paths));
}

/** 文档排版（去硬回车） */
export async function formatDocument(text: string): Promise<FormatResult> {
    return withRetry(() => getAPI().formatDocument(text));
}

/** 导出合并文档（触发 Electron 保存对话框 + 写入） */
export async function exportFile(paragraphs: string[]): Promise<{ saved_path: string }> {
    return withRetry(() => getAPI().exportFile(paragraphs));
}

/** 选择 .txt 文件（触发 Electron 打开文件对话框） */
export async function selectTxtFiles(): Promise<{ name: string; path: string; size: number }[]> {
    return withRetry(() => getAPI().selectFiles());
}
```

```rust
#[macro_use]
extern crate napi_derive;

mod document_formatter;
mod duplicate_resolver;
mod file_processor;
mod paragraph_index;
mod text_normalizer;

use document_formatter::{DocumentFormatter, FormatResult};
use duplicate_resolver::{AnalysisReport, DuplicateResolver, ExportResult, FileMeta};
use file_processor::FileProcessor;
use napi::bindgen_prelude::*;
use paragraph_index::InterFileDeduper;
use rayon::prelude::*;
use std::path::Path;
use text_normalizer::TextNormalizer;

/// 扫描并分析 TXT 文件（V3.0 napi 版本）
///
/// 完全复用 V2.0.2 的去重逻辑，仅将 `#[tauri::command]` 替换为 `#[napi]`
#[napi]
pub fn scan_files(paths: Vec<String>) -> Result<AnalysisReport> {
    // 并行读取并归一化所有文件（与 V2.0.2 lib.rs 逻辑完全相同）
    let file_results: Vec<(file_processor::FileMetadata, String, Vec<String>)> = paths
        .par_iter()
        .map(|path| {
            let normalizer = TextNormalizer::new();
            let file_path = Path::new(path);
            let file_name = file_path
                .file_name()
                .map(|n| n.to_string_lossy().to_string())
                .unwrap_or_else(|| "unknown".to_string());
            let metadata = FileProcessor::get_file_metadata(file_path)
                .map_err(|e| format!("文件 '{}' 元信息读取失败: {}", file_name, e))?;
            let content = FileProcessor::read_file_content(file_path)
                .map_err(|e| format!("文件 '{}' 读取失败: {}", file_name, e))?;
            let normalized = normalizer.normalize(&content);
            Ok::<_, String>((metadata, file_name, normalized))
        })
        .collect::<Result<Vec<_>, String>>()
        .map_err(|e| napi::Error::from_reason(format!("文件处理失败: {}", e)))?;

    // V1.1 算法：使用 InterFileDeduper 按文件顺序处理
    let mut engine = InterFileDeduper::new();
    let mut files_metadata: Vec<FileMeta> = Vec::with_capacity(file_results.len());

    for (meta, file_name, normalized_paras) in &file_results {
        files_metadata.push(FileMeta {
            file_name: file_name.clone(),
            file_size: meta.file_size,
            modified: meta.modified,
        });
        engine.process_file(file_name, normalized_paras);
    }

    let (duplicate_groups, preview_paragraphs) = engine.finalize();
    let total_files = paths.len();

    Ok(DuplicateResolver::build_report(
        duplicate_groups,
        preview_paragraphs,
        total_files,
        files_metadata,
    ))
}

/// 文档排版（去硬回车）— V3.0 napi 版本
#[napi]
pub fn format_document(text: String) -> Result<FormatResult> {
    if text.trim().is_empty() {
        return Err(napi::Error::from_reason(
            "排版处理失败: 输入文本为空".to_string(),
        ));
    }

    if text.len() > 100 * 1024 * 1024 {
        return Err(napi::Error::from_reason(
            "排版处理失败: 文本大小超过限制".to_string(),
        ));
    }

    let formatter = DocumentFormatter::new();
    Ok(formatter.format(&text))
}
```

```rust
use serde::{Deserialize, Serialize};

#[napi(object)]
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct SourceInfo {

    pub start_line: u32,   // napi-rs 不支持 usize，改用 u32

    pub snippet: String,
    pub sources: Vec<SourceInfo>,

pub struct PreviewParagraph {
    pub id: String,
    pub text: String,
    pub content_hash: String,
    pub source_files: Vec<String>,
    pub is_original: bool,

    pub file_name: String,

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ExportResult {
    pub saved_path: String,
}
```

> **关键变更**：
> - 所有 `usize` → `u32`（napi-rs 不支持平台依赖的整数类型）
> - 所有 `u64` → `u32`（同上；文件大小和修改时间在 JS 端用 number 表示即可）
> - 添加 `#[napi(object)]` 使类型可从 Rust 传递到 JS
> - 同时保留 `serde::Serialize/Deserialize`（内部测试和其他用途）

> **⚠️ 连带变更**：由于 `start_line`、`file_size`、`modified`、`occurrence_count`、`total_files` 从 `usize`/`u64` 改为 `u32`，需同步修改 `paragraph_index.rs` 和 `file_processor.rs` 中的对应赋值处——
>   - `paragraph_index.rs` 赋值 `start_line` 处：`.start_line = (line_idx + 1) as u32`
>   - `paragraph_index.rs` `occurrence_count` 处：`.occurrence_count = unique_file_count as u32`
>   - `file_processor.rs` `FileMetadata` 结构体中 `file_size` 和 `modified` 字段类型也需改为 `u32`
>   - `lib.rs` 中构建 `FileMeta` 时：`.file_size = meta.file_size as u32`、`.modified = meta.modified as u32`

---

#### 🔧 `native/src/paragraph_index.rs`（小修）

改动仅在类型赋值处的类型转换（`usize → u32`），核心算法零改动。

需修改的行（`finalize()` 方法末尾附近）：
```rust
// 改动前
duplicate_groups.push(DuplicateGroup {
    // ...
    occurrence_count: unique_file_count,    // usize
    sources: sources.clone(),               // Vec<SourceInfo { start_line: usize }>
});

// 改动后
duplicate_groups.push(DuplicateGroup {
    // ...
    occurrence_count: unique_file_count as u32,
    sources: sources.clone(),               // SourceInfo.start_line 已为 u32
});
```

```powershell
# V2.0.2 开发流程（旧）

```

### 4.2 构建发布流程

```powershell
# V2.0.2 构建（旧）
npm run build              # Vite 构建
cd src-tauri; cargo build --release  # Rust 编译
# → 输出 deploy/text-unifier.exe

# V3.0 构建（新）
npm run napi:build         # 编译 Rust → .node 文件
npm run build              # Vite 构建
npm run electron:build     # electron-builder 打包
# → 输出 release/Text Unifier_3.0.0_x64-setup.exe
# → 输出 release/TextUnifier_Portable_3.0.0.exe
```

### 4.3 包体大小预估

| 产物 | V2.0.2 | V3.0 预计 | 增幅 |

## 五、风险评估与缓解

| 风险 | 等级 | 影响 | 缓解措施 |
| :--- | :---: | :--- | :--- |

| Rust 正则编译兼容 | 🟢 极低 | regex crate 在所有平台行为一致 | Rust 跨平台稳定性已验证 |
| Electron 首次启动慢 | 🟡 低 | Chromium 冷启动约 2~3s | 可在启动时显示 splash，体验可接受 |
